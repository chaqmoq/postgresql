Package.swift

// swift-tools-version:5.1

import PackageDescription

let package = Package(
    name: "MyApp",
    dependencies: [
        .package(url: "https://github.com/chaqmoq/postgresql.git", .branch("master"))
    ],
    targets: [
        .target(name: "MyApp", dependencies: ["PostgreSQL"]),
        .testTarget(name: "MyAppTests", dependencies: ["MyApp"])
    ]
)
